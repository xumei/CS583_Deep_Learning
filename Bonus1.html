2. Federated Averaging Model¶

2.1. Worker Node¶

In [309]:
class Worker:
  def __init__(self, x, y):
    self.x = x # s-by-d local feature matrix
    self.y = y # s-vy-1 local label matrix
    self.s = x.shape[0] # number of local samples
    self.d = x.shape[1] # number of features
    self.w = numpy.zeros((d, 1)) # d-by-1 model parameter vector

  # Set the model parameters to the latest
  def set_param(self, w):
    self.w = w
  
  # Compute the local lose
  def loss(self):
    yx = numpy.multiply(self.y, self.x) # s-by-d matrix
    yxw = numpy.dot(yx, self.w) # s-by-1 matrix
    vec1 = numpy.exp(-yxw) # s-by-1 matrix
    vec2 = numpy.log(1 + vec1) # s-by-1 matrix
    return numpy.sum(vec2) # loss function

  # Compute the local gradient - GD
  def gradient(self):
    yx = numpy.multiply(self.y, self.x) # s-by-d matrix
    yxw = numpy.dot(yx, self.w) # s-by-1 matrix
    vec1 = numpy.exp(yxw) # s-by-1 matrix
    vec2 = numpy.divide(yx, 1+vec1) # s-by-d matrix
    g = -numpy.sum(vec2, axis=0).reshape(self.d, 1) # d-by-1 matrix
    return g
  
  # Compute the local gradient - SGD
  def sgd_gradient(self, xi, yi, lam):
    yx = numpy.multiply(yi, xi)
    yxw = numpy.dot(yx, self.w)
    g = -yx.T / (1 + numpy.exp(yxw))
    return g

  # Update the local parameters using gradient descent
  # Args:
  #   alpha: learning rate 
  def gd(self, alpha, lam, q):
    for _ in range(q):
      self.w -= alpha * (self.gradient() / self.s + lam * self.w)

  # Update the local parameters using SGD
  def sgd(self, alpha, lam, q):
    for _ in range(q):
      rand_indices = numpy.random.permutation(self.s)
      x_rand = self.x[rand_indices, :]
      y_rand = self.y[rand_indices]
      for i in range(self.s):
        xi = x_rand[i, :]
        yi = float(y_rand[i])
        g = self.sgd_gradient(xi, yi, lam)
        self.w -= alpha * (g + lam * self.w)
      alpha *= 0.9

2.2. Server¶

In [310]:
class Server:
  def __init__(self, m, n, d):
    self.m = m # number of worker nodes
    self.n = n # number of training samples
    self.d = d # number of features
    self.w = numpy.zeros((d, 1)) # d-by-1 model parameter vector
    self.g = numpy.zeros((d, 1)) # d-by-1 gradient
    self.loss = 0 
    self.obj = 0

  # Sum the local w and loss functions evaluated by the worker
  # Args:
  #   params: a list of d-by-1 vectors
  #   losses: a list of scalars
  def aggregate(self, params, losses):
    self.w = numpy.zeros((self.d, 1))
    self.loss = 0
    for k in range(self.m):
      self.w += params[k]
      self.loss += losses[k]

  # Update the model parameters
  def new_param(self):
      self.w = self.w / self.m
  
  # Broadcast the new parameter to workers
  def broadcast(self):
    return self.w

  # Compute the objective function  (sum of loss and regularization)
  def objective(self, lam):
    reg = lam / 2 * numpy.sum(self.w * self.w)
    self.obj = self.loss / self.n + reg
    return self.obj

2.3. Initialization¶

The following function create one server and a list of m workers. I partition the n samples to m parts and use each part to create one worker.

I use m = 4 workers and apply the function create server workers. The returned are server (an object) and workers (a list of objects).

In [311]:
import math

# Create a server and m worker nodes

def create_server_workers(m, x, y):
  n, d = x.shape
  s = math.floor(n / m)
  server = Server(m, n, d)
  workers = []

  for i in range(m):
    indices = list(range(i*s, (i+1)*s))
    worker = Worker(x[indices, :], y[indices, :])
    workers.append(worker)
  
  return server, workers

lam = 1E-6
alpha = 1e-1
max_epoch = 50

def Fagd(q):
  m = 4 # number of worker nodes
  server, workers = create_server_workers(m, x_train, y_train)
  objvals_gd = numpy.zeros(max_epoch)
  for t in range(max_epoch):
    # step1: broadcast server's parameter to workers
    w = server.broadcast()
    for worker in workers:
      worker.set_param(w)
    
    # step2: workers' local computation
    params = []
    losses = []
    for worker in workers:
      worker.gradient()
      worker.gd(alpha, lam, q)
      params.append(worker.w)
      l = worker.loss()
      losses.append(l)

    # step3: aggregate the workers' parameters
    server.aggregate(params, losses)

    # step4: server update  the model parameters
    server.new_param() # compute new w
    obj = server.objective(lam) # compute objective function
    objvals_gd[t] = obj
    print('Objective value at epoch t=' + str(t) + ' is ' + str(obj))
  return objvals_gd
